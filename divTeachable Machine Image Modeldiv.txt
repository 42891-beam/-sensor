<script type="text/javascript">
    // ลิงก์โมเดลเวอร์ชันล่าสุดของคุณ
    const URL = "https://teachablemachine.withgoogle.com/models/jkT2DnYZd/";[cite: 2]

    let model, webcam, labelContainer, maxPredictions;
    let predictTimer;

    // โหลดโมเดลและตั้งค่ากล้องเว็บแคม
    async function init() {
        // ซ่อนปุ่มเมื่อถูกคลิก เพื่อป้องกันการโหลดซ้ำซ้อน
        document.querySelector('.btn-start').style.display = 'none';

        const modelURL = URL + "model.json";[cite: 2]
        const metadataURL = URL + "metadata.json";[cite: 2]

        // โหลดโมเดล AI[cite: 2]
        model = await tmImage.load(modelURL, metadataURL);[cite: 2]
        maxPredictions = model.getTotalClasses();[cite: 2]

        // ตั้งค่าขนาดและการแสดงผลของ Webcam (ขยายขนาดขึ้นเป็น 260x260 ให้เห็นชัดเจนขึ้น)
        const flip = true; 
        webcam = new tmImage.Webcam(260, 260, flip);[cite: 2]
        await webcam.setup();[cite: 2]
        await webcam.play();[cite: 2]
        
        // สั่งให้กล้องอัปเดตภาพเคลื่อนไหวแบบ Real-time (ภาพเคลื่อนไหวลื่นไหลปกติ)
        window.requestAnimationFrame(loop);[cite: 2]

        // นำ Element กล้องไปใส่ใน HTML[cite: 2]
        document.getElementById("webcam-container").appendChild(webcam.canvas);[cite: 2]
        labelContainer = document.getElementById("label-container");[cite: 2]
        
        // สร้างกรอบสำหรับแสดงผลลัพธ์
        for (let i = 0; i < maxPredictions; i++) {[cite: 2]
            let div = document.createElement("div");
            div.className = "label-row";
            labelContainer.appendChild(div);[cite: 2]
        }

        // ทำนายผลครั้งแรกทันทีที่เปิดกล้อง
        await predict();

        // ตั้งเวลาทำงานอัตโนมัติ: สั่งประมวลผลวิเคราะห์รูปภาพทุกๆ 3 วินาที (3000 มิลลิวินาที)
        predictTimer = setInterval(predict, 3000);
    }

    // ลูปในการดึงเฟรมภาพจากกล้องมาแสดงผล (เฉพาะการแสดงภาพให้ลื่นไหล)
    async function loop() {
        webcam.update(); // อัปเดตการแสดงผลของหน้าจอกล้องแคม[cite: 2]
        window.requestAnimationFrame(loop);[cite: 2]
    }

    // ฟังก์ชันในการส่งภาพไปให้ AI ประมวลผลและแสดงความน่าจะเป็น
    async function predict() {
        if (!model || !webcam) return;

        const prediction = await model.predict(webcam.canvas);[cite: 2]
        for (let i = 0; i < maxPredictions; i++) {[cite: 2]
            const className = prediction[i].className;
            // เปลี่ยนค่าทศนิยมให้กลายเป็น % ที่เข้าใจง่าย เช่น 0.95 -> 95%
            const probabilityPercentage = (prediction[i].probability * 100).toFixed(0) + "%";
            
            // ปรับแต่งโครงสร้าง HTML ด้านในเพื่อความสวยงามแยกสัดส่วนชัดเจน
            labelContainer.childNodes[i].innerHTML = `
                <span class="label-name">${className}</span>
                <span class="label-percent">${probabilityPercentage}</span>
            `;
        }
    }
